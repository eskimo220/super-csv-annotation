77      /**
78       * 1レコード分を読み込みます。
79       * 
80       * @return Beanのレコード。読み込むレコードがない場合は、nullを返します。
81       * 
82       * @throws IOException レコードの読み込みに失敗した場合。
83       * @throws SuperCsvNoMatchColumnSizeException レコードのカラムサイズに問題がある場合
84       * @throws SuperCsvBindingException セルの値に問題がある場合
85       * @throws SuperCsvException 設定など、その他に問題がある場合
86       * 
87       */
88      public T read() throws IOException {
89          
90          if(readRow()) {
91              
92              final T bean = instantiateBean(beanMappingCache.getOriginal().getType());
93              final CsvBindingErrors bindingErrors = new CsvBindingErrors(beanMappingCache.getOriginal().getType());
94              
95              final CsvContext context = new CsvContext(getLineNumber(), getRowNumber(), 1);
96              context.setRowSource(new ArrayList<Object>(processedColumns));
97              
98              Optional<SuperCsvRowException> rowException = Optional.empty();
99              try {
100                 executeCellProcessor(processedColumns, getColumns(), beanMappingCache.getCellProcessorsForReading(), context);
101                 
102             } catch(SuperCsvRowException e) {
103                 /*
104                  * カラムごとのCellProcessorのエラーの場合、別なValidatorで値を検証するために、
105                  * 後から判定を行うようにする。
106                  */
107                 rowException = Optional.of(e);
108                 
109                 final List<CsvError> errors = exceptionConverter.convert(e, beanMappingCache.getOriginal());
110                 bindingErrors.addAllErrors(errors);
111                 
112             } catch(SuperCsvException e) {
113                 errorMessages.addAll(exceptionConverter.convertAndFormat(e, beanMappingCache.getOriginal()));
114                 throw e;
115             }
116             
117             // コールバックメソッドの実行（読み込み前）
118             for(CallbackMethod callback : beanMappingCache.getOriginal().getPreReadMethods()) {
119                 callback.invoke(bean, context, bindingErrors, beanMappingCache.getOriginal());
120             }
121             
122             // beanへのマッピング
123             populateBean(bean, beanMappingCache.getNameMapping(), bindingErrors);
124             
125             // Bean(レコード)の入力値検証
126             for(CsvValidator<T> recordValidator : validators) {
127                 recordValidator.validate(bean, bindingErrors, new ValidationContext<>(context, beanMappingCache.getOriginal()));
128             }
129             
130             // コールバックメソッドの実行（読み込み後）
131             for(CallbackMethod callback : beanMappingCache.getOriginal().getPostReadMethods()) {
132                 callback.invoke(bean, context, bindingErrors, beanMappingCache.getOriginal());
133             }
134             
135             // エラーメッセージの変換
136             processErrors(bindingErrors, context, rowException);
137             
138             return bean;
139             
140         }
141         
142         return null; // EOF
143         
144         
145     }

147     /**
148      * CSVのヘッダーの検証を行います。
149      * 
150      * @param sourceHeader オリジナルのヘッダー情報。
151      * @param definedHeader アノテーションなどの定義を元にしたヘッダー情報
152      * @throws SuperCsvNoMatchColumnSizeException ヘッダーのサイズ（カラム数）がBean定義と一致しない場合。
153      * @throws SuperCsvNoMatchHeaderException ヘッダーの値がBean定義と一致しない場合。
154      */
155     protected void validateHeader(final String[] sourceHeader, final String[] definedHeader) {
156         
157         // check column size.
158         if(sourceHeader.length != definedHeader.length) {
159             final CsvContext context = new CsvContext(1, 1, 1);
160             throw new SuperCsvNoMatchColumnSizeException(sourceHeader.length, definedHeader.length, context);
161             
162         }
163         
164         // check header value
165         for(int i=0; i < sourceHeader.length; i++) {
166             if(!sourceHeader[i].equals(definedHeader[i])) {
167                 final CsvContext context = new CsvContext(1, 1, i+1);
168                 throw new SuperCsvNoMatchHeaderException(sourceHeader, definedHeader, context);
169             }
170         }
171         
172     }

174     /**
175      * 行の例外情報をメッセージに変換したりします。
176      * @param bindingErrors
177      * @param context
178      * @param rowException
179      */
180     protected void processErrors(final CsvBindingErrors bindingErrors, final CsvContext context,
181             final Optional<SuperCsvRowException> rowException) {
182         if(bindingErrors.hasErrors()) {
183             final List<String> message = bindingErrors.getAllErrors().stream()
184                     .map(error -> error.format(exceptionConverter.getMessageResolver(), exceptionConverter.getMessageInterpolator()))
185                     .collect(Collectors.toList());
186             errorMessages.addAll(message);
187             
188             final SuperCsvBindingException bindingException = new SuperCsvBindingException("has binding error.", context, bindingErrors);
189             rowException.ifPresent(re -> bindingException.addAllProcessingErrors(re.getColumnErrors()));
190             
191             throw bindingException;
192         }
193     }

196      * 指定したBeanのクラスのインスタンスを作成する。
197      * 
198      * @param clazz Beanのクラスタイプ。
199      * @return Beanのインスタンス。
200      * @throws SuperCsvReflectionException Beanのインスタンスの作成に失敗した場合。
201      */
202     protected T instantiateBean(final Class<T> clazz) {
203         
204         final T bean;
205         if( clazz.isInterface() ) {
206             bean = BeanInterfaceProxy.createProxy(clazz);
207         } else {
208             try {
209                 bean = clazz.newInstance();
210             } catch(InstantiationException e) {
211                 throw new SuperCsvReflectionException(String.format(
212                     "error instantiating bean, check that %s has a default no-args constructor", clazz.getName()), e);
213             } catch(IllegalAccessException e) {
214                 throw new SuperCsvReflectionException("error instantiating bean", e);
215             }
216         }
217         
218         return bean;
219         
220     }

222     /**
223      * 行の各カラムの値に対して、CellProcessorを適用します。
224      * @param destination
225      * @param source
226      * @param processors
227      * @param context
228      * @throws SuperCsvNoMatchColumnSizeException カラムサイズが定義と一致しない場合
229      * @throws SuperCsvRowException CellProcessor内で発生した例外
230      */
231     protected void executeCellProcessor(final List<Object> destination, final List<String> source,
232             final CellProcessor[] processors, final CsvContext context) {
233         
234         if(source.size() != processors.length) {
235             throw new SuperCsvNoMatchColumnSizeException(source.size(), processors.length, context);
236             
237         }
238         
239         destination.clear();
240         
241         final SuperCsvRowException rowException = new SuperCsvRowException(
242                 String.format("row (%d) has errors column", context.getRowNumber()), context);
243         
244         for( int i = 0; i < source.size(); i++ ) {
245             
246             try {
247                 context.setColumnNumber(i + 1); // update context (columns start at 1)
248                 
249                 if( processors[i] == null ) {
250                     destination.add(source.get(i)); // no processing required
251                 } else {
252                     destination.add(processors[i].execute(source.get(i), context)); // execute the processor chain
253                 }
254             } catch(SuperCsvCellProcessorException e) {
255                 rowException.addError(e);
256                 
257                 // 各カラムでエラーがあっても、後の入力値検証で処理を続けるために、仮に値を設定する。
258                 destination.add(source.get(i));
259                 
260             } catch(SuperCsvException e) {
261                 
262                 throw e;
263                 
264             }
265         }
266         
267         if(rowException.isNotEmptyColumnErrors()) {
268             throw rowException;
269         }
270         
271     }

273     /**
274      * Beanの各フィールドに対して値を設定する。
275      * @param resultBean
276      * @param nameMapping
277      * @param bindingErrors
278      */
279     protected void populateBean(final T resultBean, final String[] nameMapping, final CsvBindingErrors bindingErrors) {
280         
281         // map each column to its associated field on the bean
282         for( int i = 0; i < nameMapping.length; i++ ) {
283             final String fieldName = nameMapping[i];
284             final Object fieldValue = processedColumns.get(i);
285             
286             // don't call a set-method in the bean if there is no name mapping for the column or no result to store
287             if( fieldName == null || fieldValue == null || bindingErrors.hasFieldErrors(fieldName)) {
288                 continue;
289             }
290             
291             // invoke the setter on the bean
292             final Method setMethod = cache.getSetMethod(resultBean, fieldName, fieldValue.getClass());
293             try {
294                 setMethod.invoke(resultBean, fieldValue);
295                 
296             } catch(final Exception e) {
297                 throw new SuperCsvReflectionException(String.format("error invoking method %s()", setMethod.getName()), e);
298             }
299             
300         }
301         
302     }
