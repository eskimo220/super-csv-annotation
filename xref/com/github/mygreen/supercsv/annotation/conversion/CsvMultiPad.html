16  /**
17   * パディングするためのアノテーションです。
18   * <p>アノテーション {@link CsvLeftPad}や{@link CsvRightPad}より、柔軟に指定ができます。</p>
19   *
20   * <h3 class="description">基本的な使い方</h3>
21   * 
22   * <ul>
23   *   <li>属性 {@link #size()}で、パディングするサイズを指定します。
24   *     <br>変換対象の文字長が属性{@link #size()}の値を超える場合、パディングは行いません。
25   *   </li>
26   *   <li>属性{@link #padChar()}でパディング文字を指定することができます。
27   *     <br>デフォルトでは、半角空白がパディング文字です。
28   *   </li>
29   *   <li>文字サイズは、考え型によって変わるため、{@link #paddingProcessor()} で処理方式を指定します。
30   *      <br>例えば、{@link CharWidthPaddingProcessor}は、文字の幅（半角の長さ=1、全角の長さ=2）として書き込み時にパディングします。
31   *   </li>
32   *   <li>アノテーション{@link CsvTrim}と併用する際には、処理結果が互いに変換対象となるため、
33   *      属性{@link #cases()}や{@link #groups()}で適用するケースを分けるようにしてください。
34   *   </li>
35   * </ul>
36   * 
37   * <pre class="highlight"><code class="java">
38   * {@literal @CsvBean}
39   * public class SampleCsv {
40   *     
41   *     // 値が[10]の場合、結果は、[&nbsp;&nbsp;&nbsp;10]となります。
42   *     {@literal @CsvColumn(number=1)}
43   *     {@literal @CsvMultiPad(size=5)}
44   *     private Integer id;
45   *     
46   *     // パディング文字を変更する場合
47   *     {@literal @CsvColumn(number=2)}
48   *     {@literal @CsvMultiPad(size=5, padChar='_')}
49   *     private Integer number;
50   *     
51   *     // パディングする際の処理方法を指定する場合
52   *     {@literal @CsvColumn(number=2)}
53   *     {@literal @CsvMultiPad(size=5, padChar='　', paddingProcessor=CharWidthPaddingProcessor.class)}
54   *     private String name;
55   *     
56   *     // 書き込み時のみ適用する場合
57   *     {@literal @CsvColumn(number=4)}
58   *     {@literal @CsvTrim(cases=BuildCase.Read)}
59   *     {@literal @CsvMultiPad(size=20, cases=BuildCase.Write)}
60   *     private String comment;
61   *     
62   *     // getter/setterは省略
63   * }
64   * </code></pre>
65   *
66   * @since 2.1
67   * @author T.TSUCHIE
68   *
69   */
