19  /**
20   * 固定長のカラムを表現するためのアノテーションです。
21   * <p>このアノテーションは、次の合成のアノテーションによって、構成されています。</p>
22   * <ul>
23   *  <li>書き込み時は、アノテーション{@link CsvMultiPad}によって、パディングします。</li>
24   *  <li>読み込み時は、アノテーション{@link CsvOneSideTrim}によって、トリムします。</li>
25   * </ul>
26   * 
27   * <h3 class="description">基本的な使い方</h3>
28   * 
29   * <ul>
30   *   <li>属性{@link #size()}で文字のサイズを指定します。</li>
31   *   <li>文字のサイズは、カウント方法の考え型によって変わるため、{@link #paddingProcessor()} で処理方式を指定します。
32   *      <br>例えば、{@link CharWidthPaddingProcessor}は、文字の幅（半角文字のサイズ=1、全角文字のサイズ=2）として書き込み時にパディングします。
33   *   </li>
34   *   <li>属性{@link #rightAlign()}によって、右寄せか左寄せか指定します。
35   *    <br>trueのとき右寄せ、falseのとき左よせで、デフォルトは左寄せです。
36   *   </li>
37   * </ul>
38   * 
39   * <pre class="highlight"><code class="java">
40   * {@literal @CsvBean}
41   * public class SampleCsv {
42   *     
43   *     // 右詰めする
44   *     {@literal @CsvColumn(number=1)}
45   *     {@literal @CsvFixedSize(size=10, rightAlign=true)}
46   *     private Integer value;
47   *     
48   *     // 全角空白で埋める。
49   *     // ただし、文字の幅（半角のサイズ=1、全角のサイズ=2）として処理する。
50   *     {@literal @CsvColumn(number=2)}
51   *     {@literal @CsvFullChar}
52   *     {@literal @CsvFixedSize(size=20, padChar='　', paddingProcessor=CharWidthPaddingProcessor.class)}
53   *     private String userName;
54   *     
55   *     // 他のオブジェクト型への変換を行う
56   *     {@literal @CsvColumn(number=3)}
57   *     {@literal @CsvFixedSize(size=10, padChar='_')}
58   *     {@literal @CsvDateTimeFormat(pattern="uuuu-MM-dd")}
59   *     private LocalDate birthday;
60   *     
61   *     // 指定した文字長を超えた場合、切り落とす。
62   *     {@literal @CsvColumn(number=4, label="コメント")}
63   *     {@literal @CsvFixedSize(size=20, chopped=true)}
64   *     private String comment;
65   *     
66   *     // getter/setterは省略
67   * }
68   * </code></pre>
69   * 
70   * @since 2.1
71   * @author T.TSUCHIE
72   *
73   */
74  @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
75  @Retention(RetentionPolicy.RUNTIME)
76  @Documented
77  @Repeatable(CsvFixedSize.List.class)
78  @CsvComposition
79  @CsvMultiPad(size=0, cases=BuildCase.Write, order=1000)
80  @CsvOneSideTrim(cases=BuildCase.Read)
81  public @interface CsvFixedSize {
82      
83      /**
84       * カラムのサイズを指定します。
85       * <p>値は1以上を指定する必要があります。</p>
86       * <p>サイズは考え型によってバイト数、文字幅など異なるため、
87       *    属性{@link #paddingProcessor()}によって、変更することができます。
88       * </p>
89       * 
90       * @return カラムのサイズ。
91       */
92      @CsvOverridesAttribute(annotation=CsvMultiPad.class, name="size")
93      int size();
94      
95      /**
96       * パディングする際の文字を指定します。
97       * <br>読み込み時には、トリム対象の文字となります。
98       *
99       * @return パディング文字。
100      */
101     @CsvOverridesAttribute(annotation=CsvMultiPad.class, name="padChar")
102     @CsvOverridesAttribute(annotation=CsvOneSideTrim.class, name="trimChar")
103     char padChar() default ' ';
104     
105     /**
106      * パディング時に指定したカラムの長さを超えた場合、切り出すかどうか指定します。
107      * @return trueの場合、指定したカラムの長さを超えた場合切り出します。
108      *
109      */
110     @CsvOverridesAttribute(annotation=CsvMultiPad.class, name="chopped")
111     boolean chopped() default false;
112     
113     /**
114      * パディングの処理方法を指定します。
115      * <p>{@link PaddingProcessor}を実装したクラス指定します。
116      *  <br>標準では次の実装が提供されています。
117      * </p>
118      * <ul>
119      *   <li>{@link SimplePaddingProcessor}：文字の種別にかかわらず１文字としてカウントしてパディングします。</li>
120      *   <li>{@link CharWidthPaddingProcessor}：文字の幅（半角は1文字、全角は2文字）によってカウントしてパディングします。</li>
121      *   <li>{@link ByteSizePaddingProcessor}：バイト数によってカウントしてパディングします。</li>
122      * </ul>
123      *
124      * @return パディング処理の実装クラス。
125      */
126     @CsvOverridesAttribute(annotation=CsvMultiPad.class, name="paddingProcessor")
127     Class<? extends PaddingProcessor> paddingProcessor() default CharWidthPaddingProcessor.class;
128     
129     /**
130      * 右寄せをするかどうか指定します。
131      * <br>読み込み時のトリム時は、逆の意味になるので注意。
132      * <br>falseの場合は、左詰めです。
133      *
134      * @return パディング文字が右側にあるかどうか。
135      */
136     @CsvOverridesAttribute(annotation=CsvMultiPad.class, name="rightAlign")
137     @CsvOverridesAttribute(annotation=CsvOneSideTrim.class, name="leftAlign")
138     boolean rightAlign() default false;
139     
140     /**
141      * グループのクラスを指定します。
142      * <p>処理ごとに適用するアノテーションを切り替えたい場合に指定します。
143      * @return 指定しない場合は、{@link DefaultGroup}が適用され全ての処理に適用されます。
144      */
145     Class<?>[] groups() default {};
146     
147     @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
148     @Retention(RetentionPolicy.RUNTIME)
149     @Documented
150     @interface List {
151         
152         CsvFixedSize[] value();
153     }
