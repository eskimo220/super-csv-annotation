17  /**
18   * CSVのBeanであることを表現するためのアノテーションです。
19   * <p>クラスに付与します。</p>
20   * 
21   * <h3 class="description">基本的な使い方</h3>
22   * CSVをマッピングするBeanのクラスに付与します。
23   * 
24   * <pre class="highlight"><code class="java">
25   * {@literal @CsvBean}
26   * public class SampleCsv {
27   *     
28   *     {@literal @CsvColumn(number=1)}
29   *     private int no;
30   *     
31   *     {@literal @CsvColumn(number=2, label="名前")}
32   *     private String name;
33   *     
34   *     // getter/setterは省略
35   * }
36   * </code></pre>
37   * 
38   * @version 2.1
39   * @author T.TSUCHIE
40   *
41   */
42  @Target({ElementType.TYPE})
43  @Retention(RetentionPolicy.RUNTIME)
44  @Documented
45  public @interface CsvBean {
46      
47      /**
48       * ヘッダーが存在することを前提として処理します。
49       * <p>{@link CsvAnnotationBeanReader#readAll(boolean)}、{@link CsvAnnotationBeanWriter#writeAll(java.util.Collection, boolean)}を呼び出した際に利用します。</p>
50       * @return trueの場合、読み込み時には1行目はヘッダー情報として読み込み、書き出し時にはヘッダー行が出力されるようになります。
51       */
52      boolean header() default false;
53      
54      /**
55       * ヘッダー行の読み込み時に、値の検証を行うか指定します。
56       * <p>部分的にカラムの読み込みを行う際、アノテーション{@link CsvPartial}で省略した見出しを定義していない場合は、属性の値をfalseに設定してください。</p>
57       * 
58       * @since 2.0
59       * @return trueの場合、ヘッダー行の値が定義されている値と同じかどうか検証を行います。
60       *         ヘッダーの値が不正な場合、例外{@link SuperCsvNoMatchHeaderException}がスローされます。
61       */
62      boolean validateHeader() default false;
63      
64      /**
65       * カラムに対するヘッダーラベルを取得方法を指定します。
66       * 
67       * @since 2.1
68       * @return {@link HeaderMapper}の実装クラスを指定します。
69       */
70      Class<? extends HeaderMapper> headerMapper() default DefaultHeaderMapper.class;
71      
72      /**
73       * レコードに対する値の検証を行うクラスを指定します。
74       * <p>カラム間の相関チェックやBean Validationを使用する際に指定します。</p>
75       * 
76       * @since 2.0
77       * @return {@link CsvValidator}を実装したクラスを指定します。
78       *         複数指定可能で、指定した順に実行されます。
79       * 
80       */
81      Class<? extends CsvValidator<?>>[] validators() default {};
82      
83      /**
84       * ライフサイクルコールバック用のリスナークラスを指定するためのアノテーション。
85       * @since 2.0
86       * @return コールバック用アノテーションを指定したメソッドを指定します。
87       */
88      Class<?>[] listeners() default {};
89      
90  }
