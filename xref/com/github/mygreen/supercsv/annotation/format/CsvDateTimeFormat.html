113  * <h3 class="description">基本的な使い方</h3>
114  * 
115  * <ul>
116  *   <li>属性{@link #pattern()}で、書式を指定します。
117  *     <br>省略した場合は、標準の書式が適用されます。
118  *   </li>
119  *   <li>属性{@link #locale()}でロケールを指定します。
120  *     <ul>
121  *       <li>言語コードのみを指定する場合、{@literal ja} の2桁で指定します。</li>
122  *       <li>言語コードと国コードを指定する場合、{@literal ja_JP} のようにアンダーバーで区切り指定します。</li>
123  *       <li>和暦を扱う時など、バリアントを指定する場合も同様に、{@literal ja_JP_JP} のようにさらにアンダーバーで区切り指定します。</li>
124  *     </ul>
125  *    </li>
126  * </ul>

136  *     
137  *     {@literal @CsvColumn(number=2, label="更新日時")}
138  *     {@literal @CsvDateTimeFormat(pattern="uuuu/MM/dd HH:mm:ss")}

140  *     
141  *     // getter/setterは省略
142  * }

149 @Target({ElementType.FIELD, ElementType.ANNOTATION_TYPE})
150 @Retention(RetentionPolicy.RUNTIME)
151 @Documented
152 public @interface CsvDateTimeFormat {
153     
154     /**
155      * 日時の書式を指定します。
156      * <p>クラスタイプごとに、指定可能な書式は異なります。</p>
157      * <p>{@literal java.util.Date/java.util.Calendar/java.sql.Date/java.sql.Time/java.sql.Timestamp}の場合、
158      *   {@link java.text.SimpleDateFormat}で解釈可能な書式を指定します。
159      * </p>
160      * <p>{@literal java.time.LocalDateTime/java.time.LocalDate/java.time.LocalTime/java.time.ZonedDateTime/java.time.OffsetDateTime/java.time.OffsetTime/java.time.Year/java.time,YearMonth/java.time/MonthDay}の場合、
161      *   {@link java.time.format.DateTimeFormatter}で解釈可能な書式を指定します。
162      * </p>
163      * <p>{@literal org.joda.time.LocalDateTime/org.joda.time.LocalDate/org.joda.time.LocalTime/org.joda.time.YearMonth/org.joda.time.MonthDay}の場合、
164      *   {@link org.joda.time.format.DateTimeFormat}で解釈可能な書式を指定します。
165      * </p>
166      * @return 指定しない場合は、クラスタイプごとの標準の書式が適用されます。
167      */
168     String pattern() default "";
169     
170     /**
171      * 読み込み時に日時の解析を曖昧に行うか指定します。
172      * <p>曖昧に解析する場合、例えば、{@literal 2016-02-31} と存在しない日を解析すると、{@literal 2016-03-02} と自動的に補正が行われます。</p>
173      * @return trueの場合、曖昧に解析を行います。
174      */
175     boolean lenient() default false;
176     
177     /**
178      * タイムゾーンを指定します。
179      * <p>{@link TimeZone#getTimeZone(String)}で解釈可能な値を指定する必要があります。</p>
180      * <p>{@literal Asia/Tokyo, GMT, GMT+09:00}などの値を指定します。</p>
181      * <p>ただし、オフセットを持たないクラスタイプ{@literal LocalDateTime, LocalDate, LocalTime}の時は、指定しても意味がありません。</p>
182      * @return 省略した場合、システム標準の値を使用します。
183      */
184     String timezone() default "";
185     
186     /**
187      * ロケールを指定します。
188      * <p>{@literal <言語コード>}、{@literal <言語コード>_<国コード>}、{@literal <言語コード>_<国コード>_<バリアント>}の3つの何れかで書式を指定します。</p>
189      * <p>例 'ja'、'ja_JP'、'ja_JP_JP'</p>
190      * @return 省略した場合、システム標準の値を使用します。
191      */
192     String locale() default "";
193     
194     /**
195      * エラー時のメッセージを指定します。
196      * <p>{@literal ${key}}の書式の場合、プロパティファイルから取得した値を指定できます。</p>
197      * <p>このメッセージは、文字列を日時型にパースする際に失敗したときに適用されるメッセージに使用します。</p>
198      * 
199      * <ul>
200      *   <li>lineNumber : カラムの値に改行が含まれている場合を考慮した実際の行番号です。1から始まります。</li>
201      *   <li>rowNumber : CSVの行番号です。1から始まります。</li>
202      *   <li>columnNumber : CSVの列番号です。1から始まります。</li>
203      *   <li>label : カラムの見出し名です。</li>
204      *   <li>validatedValue : 実際のカラムの値です。</li>
205      *   <li>pattern : アノテーションの属性{@link #pattern()}の値です。指定されていない場合はデフォルトの値が設定されます。</li>
206      * </ul>
207      * 
208      * @return 省略した場合は、適用された{@link CellProcessor}に基づいたメッセージが出力されます。
209      */
210     String message() default "{com.github.mygreen.supercsv.annotation.format.CsvDateTimeFormat.message}";
